var SCORE_API='https://site.api.espn.com/apis/site/v2/sports/basketball/mens-college-basketball/scoreboard';
var liveScoreActive=false;
var scoreInterval=null;

function getScoreURLs(){
  // Fetch both tournament days to cover all our games
  return [
    SCORE_API+'?dates=20260319&groups=100&limit=50',
    SCORE_API+'?dates=20260320&groups=100&limit=50'
  ];
}

function nameMatch(espnName,ourNames){
  var n=espnName.toLowerCase();
  for(var i=0;i<ourNames.length;i++){
    var m=ourNames[i].toLowerCase();
    if(n.indexOf(m)>=0||m.indexOf(n)>=0) return true;
    // Handle partial matches
    if(m==='mcneese'&&n.indexOf('mcneese')>=0) return true;
    if(m==='liu'&&(n.indexOf('long island')>=0||n.indexOf('liu')>=0)) return true;
    if(m==='howard'&&n.indexOf('howard')>=0) return true;
    if(m==='miami'&&n.indexOf('miami')>=0&&n.indexOf('ohio')<0) return true;
    if(m==="saint mary's"&&n.indexOf('saint mary')>=0) return true;
    if(m==="st. john's"&&(n.indexOf('st. john')>=0||n.indexOf('saint john')>=0)) return true;
  }
  return false;
}

function fetchScores(){
  var urls=getScoreURLs();
  Promise.all(urls.map(function(u){
    return fetch(u).then(function(r){return r.json();}).catch(function(){return {events:[]};});
  })).then(function(results){
    if(!liveScoreActive){
      liveScoreActive=true;
      renderGames();
    }
    var allEvents=[];
    results.forEach(function(data){
      if(data.events) allEvents=allEvents.concat(data.events);
    });
    var changed=false;
    allEvents.forEach(function(ev){
      var comps=ev.competitions&&ev.competitions[0];
      if(!comps) return;
      var teams=comps.competitors||[];
      if(teams.length<2) return;
      var home=teams.find(function(t){return t.homeAway==='home';});
      var away=teams.find(function(t){return t.homeAway==='away';});
      if(!home||!away) return;

      var homeName=home.team.displayName||home.team.shortDisplayName||home.team.name||'';
      var awayName=away.team.displayName||away.team.shortDisplayName||away.team.name||'';
      var homeScore=parseInt(home.score)||0;
      var awayScore=parseInt(away.score)||0;
      var status=comps.status||{};
      var state=status.type?status.type.name:'';
      var clock=status.displayClock||'';
      var period=status.period||0;

      GAMES.forEach(function(g,idx){
        if(g.result==='W'||g.result==='L') return;
        var ourTeamName=g.espn[0];
        var oppTeamName=g.espn[1];
        var ourIsHome=nameMatch(homeName,[ourTeamName]);
        var ourIsAway=nameMatch(awayName,[ourTeamName]);
        var oppIsHome=nameMatch(homeName,[oppTeamName]);
        var oppIsAway=nameMatch(awayName,[oppTeamName]);

        if((ourIsHome&&oppIsAway)||(ourIsAway&&oppIsHome)){
          var ourScore=ourIsHome?homeScore:awayScore;
          var oppScore=ourIsHome?awayScore:homeScore;

          if(state==='STATUS_FINAL'){
            g.result=ourScore>oppScore?'W':'L';
            g.score=ourScore+'-'+oppScore;
            g.payout=g.result==='W'?85.82:0;
            g.time='FINAL';
            g.live=null;
            changed=true;
            saveGameState();
          } else if(state==='STATUS_IN_PROGRESS'){
            g.live={our:ourScore,opp:oppScore,clock:clock,period:period};
            g.score=ourScore+'-'+oppScore;
            changed=true;
          } else if(state==='STATUS_HALFTIME'){
            g.live={our:ourScore,opp:oppScore,clock:'HALF',period:2};
            g.score=ourScore+'-'+oppScore;
            changed=true;
          }
        }
      });
    });
    if(changed){
      renderGames();
      renderSummary();
    }
  }).catch(function(err){
    console.log('Score fetch failed:',err.message);
  });
}

function startScorePolling(){
  fetchScores();
  scoreInterval=setInterval(fetchScores,30000); // every 30 seconds
}

// ===== BRACKET STATE =====
function initBracket(){
  brkState={};brkWinnings=0;
  ['East','South','West','Midwest'].forEach(function(rgn){
    var games=BRACKET[rgn];
    brkState[rgn]={
      r64:games.map(function(g){return{a:g[0],b:g[1],winner:null};}),
      r32:[null,null,null,null],s16:[null,null],e8:null,winner:null
    };
  });
  brkState.ff={s1:null,s2:null,champ:null};
}

function advanceTeam(rgn,round,gi,pick){
  var s=brkState[rgn];
  if(round==='r64'){
    // Toggle: if this team is already the winner, undo it
    if(s.r64[gi].winner&&s.r64[gi].winner[0]===pick[0]){
      s.r64[gi].winner=null;
      var ri=Math.floor(gi/2),sl=gi%2;
      if(s.r32[ri]){s.r32[ri][sl===0?'a':'b']=null;s.r32[ri].winner=null;clearDown(rgn,'r32',ri);}
    } else {
      s.r64[gi].winner=pick;
      var ri=Math.floor(gi/2),sl=gi%2;
      if(!s.r32[ri])s.r32[ri]={a:null,b:null,winner:null};
      s.r32[ri][sl===0?'a':'b']=pick;
      s.r32[ri].winner=null;clearDown(rgn,'r32',ri);
    }
  } else if(round==='r32'){
    if(s.r32[gi].winner&&s.r32[gi].winner[0]===pick[0]){
      s.r32[gi].winner=null;
      var si=Math.floor(gi/2),sl=gi%2;
      if(s.s16[si]){s.s16[si][sl===0?'a':'b']=null;s.s16[si].winner=null;clearDown(rgn,'s16',si);}
    } else {
      s.r32[gi].winner=pick;
      var si=Math.floor(gi/2),sl=gi%2;
      if(!s.s16[si])s.s16[si]={a:null,b:null,winner:null};
      s.s16[si][sl===0?'a':'b']=pick;
      s.s16[si].winner=null;clearDown(rgn,'s16',si);
    }
  } else if(round==='s16'){
    if(s.s16[gi].winner&&s.s16[gi].winner[0]===pick[0]){
      s.s16[gi].winner=null;
      if(s.e8){s.e8[gi===0?'a':'b']=null;s.e8.winner=null;}
      s.winner=null;clearFF();
    } else {
      s.s16[gi].winner=pick;
      if(!s.e8)s.e8={a:null,b:null,winner:null};
      s.e8[gi===0?'a':'b']=pick;
      s.e8.winner=null;s.winner=null;clearFF();
    }
  } else if(round==='e8'){
    if(s.e8.winner&&s.e8.winner[0]===pick[0]){
      s.e8.winner=null;s.winner=null;clearFF();
    } else {
      s.e8.winner=pick;s.winner=pick;updateFF();
    }
  }
  calcW();renderBrkPnl();renderBrk();
}

function clearDown(rgn,fr,idx){
  var s=brkState[rgn];
  if(fr==='r32'){var si=Math.floor(idx/2);if(s.s16[si])s.s16[si].winner=null;if(s.e8)s.e8.winner=null;s.winner=null;clearFF();}
  else if(fr==='s16'){if(s.e8)s.e8.winner=null;s.winner=null;clearFF();}
}
function clearFF(){brkState.ff={s1:null,s2:null,champ:null};}
function updateFF(){
  var ff=brkState.ff;ff.s1=null;ff.s2=null;ff.champ=null;
  var sw=brkState.South.winner,ew=brkState.East.winner;
  var ww=brkState.West.winner,mw=brkState.Midwest.winner;
  if(sw&&ew)ff.s1={a:sw,b:ew,winner:null};
  if(ww&&mw)ff.s2={a:ww,b:mw,winner:null};
}
function advanceFF(semi,pick){
  var ff=brkState.ff;
  if(semi===1&&ff.s1){
    if(ff.s1.winner&&ff.s1.winner[0]===pick[0]){ff.s1.winner=null;}
    else{ff.s1.winner=pick;}
    ff.champ=null;
  }
  if(semi===2&&ff.s2){
    if(ff.s2.winner&&ff.s2.winner[0]===pick[0]){ff.s2.winner=null;}
    else{ff.s2.winner=pick;}
    ff.champ=null;
  }
  calcW();renderBrkPnl();renderBrk();
}
function advanceChamp(pick){
  var ff=brkState.ff;
  if(ff.champ&&ff.champ[0]===pick[0]){ff.champ=null;}
  else{ff.champ=pick;}
  calcW();renderBrkPnl();renderBrk();
}
function clearBracket(){initBracket();calcW();renderBrkPnl();renderBrk();}

function calcW(){
  brkWinnings=0;
  ['East','South','West','Midwest'].forEach(function(rgn){
    var s=brkState[rgn];
    s.r64.forEach(function(g){if(g.winner&&isMine(g.winner[0]))brkWinnings+=PAYOUTS.r64;});
    s.r32.forEach(function(g){if(g&&g.winner&&isMine(g.winner[0]))brkWinnings+=PAYOUTS.r32;});
    s.s16.forEach(function(g){if(g&&g.winner&&isMine(g.winner[0]))brkWinnings+=PAYOUTS.s16;});
    if(s.e8&&s.e8.winner&&isMine(s.e8.winner[0]))brkWinnings+=PAYOUTS.e8;
  });
  var ff=brkState.ff;
  if(ff.s1&&ff.s1.winner&&isMine(ff.s1.winner[0]))brkWinnings+=PAYOUTS.ff;
  if(ff.s2&&ff.s2.winner&&isMine(ff.s2.winner[0]))brkWinnings+=PAYOUTS.ff;
  if(ff.champ&&isMine(ff.champ[0]))brkWinnings+=PAYOUTS.champ;
}

  h+='<div class="brk-ff-center">';
  h+='<div class="brk-ff-label">Final Four</div>';
  h+='<div class="brk-ff-semis">';

  // Semi 1: South vs East
  h+='<div class="brk-ff-game"><div class="brk-ff-game-label">South vs East</div>';
  if(ff.s1){
    var wa=ff.s1.winner&&ff.s1.a&&ff.s1.winner[0]===ff.s1.a[0],wb=ff.s1.winner&&ff.s1.b&&ff.s1.winner[0]===ff.s1.b[0];
    h+=ff.s1.a?slotAdv(ff.s1.a,wa?'pk':'','data-ff="1|'+esc(ff.s1.a[0])+'|'+ff.s1.a[1]+'"'):slot(null,'','');
    h+=ff.s1.b?slotAdv(ff.s1.b,wb?'pk':'','data-ff="1|'+esc(ff.s1.b[0])+'|'+ff.s1.b[1]+'"'):slot(null,'','');
  } else {h+=slot(null,'','');h+=slot(null,'','');}
  h+='</div>';
  h+='<div class="brk-ff-vs">VS</div>';

  // Semi 2: West vs Midwest
  h+='<div class="brk-ff-game"><div class="brk-ff-game-label">West vs Midwest</div>';
  if(ff.s2){
    var wa=ff.s2.winner&&ff.s2.a&&ff.s2.winner[0]===ff.s2.a[0],wb=ff.s2.winner&&ff.s2.b&&ff.s2.winner[0]===ff.s2.b[0];
    h+=ff.s2.a?slotAdv(ff.s2.a,wa?'pk':'','data-ff="2|'+esc(ff.s2.a[0])+'|'+ff.s2.a[1]+'"'):slot(null,'','');
    h+=ff.s2.b?slotAdv(ff.s2.b,wb?'pk':'','data-ff="2|'+esc(ff.s2.b[0])+'|'+ff.s2.b[1]+'"'):slot(null,'','');
  } else {h+=slot(null,'','');h+=slot(null,'','');}
  h+='</div></div>';

  // Championship
  h+='<div class="brk-champ-box"><div class="brk-champ-label">Championship</div>';
  if(ff.s1&&ff.s1.winner&&ff.s2&&ff.s2.winner){
    var c1=ff.champ&&ff.champ[0]===ff.s1.winner[0],c2=ff.champ&&ff.champ[0]===ff.s2.winner[0];
    h+=slotAdv(ff.s1.winner,c1?'pk':'','data-champ="'+esc(ff.s1.winner[0])+'|'+ff.s1.winner[1]+'"');
    h+=slotAdv(ff.s2.winner,c2?'pk':'','data-champ="'+esc(ff.s2.winner[0])+'|'+ff.s2.winner[1]+'"');
  } else {h+=slot(null,'','');h+=slot(null,'','');}
  if(ff.champ){
    h+='<div class="brk-champ-winner">'+ff.champ[0]+'</div>';
  }
  h+='</div>';
  h+='</div>';

  h+='</div>';
  el.innerHTML=h;
}
